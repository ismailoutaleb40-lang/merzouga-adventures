    <script>
        // Set date min to today
        const dt = new Date().toISOString().split('T')[0];
        document.getElementById('booking-date').setAttribute('min', dt);

        const btn = document.getElementById('mobile-menu-btn');
        const links = document.getElementById('nav-links');
        
        btn.onclick = () => {
            links.classList.toggle('hidden');
            links.classList.toggle('mobile-nav-active');
        };

        function handleBooking(e) {
            e.preventDefault();
            const m = document.getElementById('successModal');
            const c = document.getElementById('modalContent');
            m.classList.replace('hidden', 'flex');
            setTimeout(() => c.classList.replace('scale-95', 'scale-100'), 50);
        }

        function closeModal() {
            const m = document.getElementById('successModal');
            const c = document.getElementById('modalContent');
            c.classList.replace('scale-100', 'scale-95');
            setTimeout(() => m.classList.replace('flex', 'hidden'), 200);
        }

        window.addEventListener('scroll', () => {
            const n = document.querySelector('nav');
            if (window.scrollY > 40) {
                n.classList.add('shadow-lg');
            } else {
                n.classList.remove('shadow-lg');
            }
        });
